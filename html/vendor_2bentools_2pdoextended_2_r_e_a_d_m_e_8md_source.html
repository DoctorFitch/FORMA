1 PDO Extended
2 ===========
3 
4 A PDO extension that will help your developer's life.
5 
6 Features :
7 - Several methods like sqlArray(), sqlrow(), sqlColumn(), sqlValue() that will help you to quickly retrieve a resultset
8 - You can disconnect, reconnect, pause the connection.
9 - Bind your values easily, example :
10 
11 ```php
12 $ids = $cnx->sqlColumn("SELECT Id FROM table WHERE CreationDate BETWEEN ? AND ?", ['2013-01-01', '2013-06-01']);
13 $id = $cnx->sqlValue("SELECT Id FROM table WHERE Id = ?", [35]);
14 $id = $cnx->sqlValue("SELECT Id FROM table WHERE Id = ?", 35); // You don't need an array if you just have 1 value to bind
15 $row = $cnx->sqlRow("SELECT * FROM table WHERE Name LIKE :Name AND Id >= :Id", ['Name' => 'foo%', 'Id' => 22]);
16 ```
17 
18 - Bind, Run and Debug your statements very easily :
19 
20 ```php
21 
22 $stmt = $cnx->prepare("SELECT Id FROM table WHERE CreationDate BETWEEN ? AND ?");
23 $ids = $stmt->sqlColumn(['2013-01-01', '2013-06-01']); // Fetches Ids in an indexed array
24 var_dump($stmt->debug()->preview()); // SELECT Id FROM table WHERE CreationDate BETWEEN '2013-01-01' AND '2013-06-01'
25 var_dump($stmt->debug()->duration); // 0.004
26 
27 ```
28 
29 - Create placeholders on the fly :
30 
31 ```php
32 $ids = [2, 5, 8, 24, 26];
33 try {
34  $rows = $cnx->sqlArray("SELECT * FROM table WHERE Id IN (".PDOStatementExtended::PlaceHolders($ids).")", $ids);
35 }
36 catch (StmtException $e) {
37  var_dump($e->getStmt()->queryString); // SELECT * FROM table WHERE Id IN (?, ?, ?, ?, ?)
38  var_dump($e->getStmt()->preview()); // SELECT * FROM table WHERE Id IN (2, 5, 8, 24, 26)
39 }
40 
41 // Also works with strings with automatic quote wrapping
42 $dates = ['2013-01-01', '2013-06-01'];
43 try {
44  $rows = $cnx->sqlArray("SELECT * FROM table WHERE CreationDate IN (".PDOStatementExtended::PlaceHolders($dates).")", $dates);
45 }
46 catch (StmtException $e) {
47  var_dump($e->getStmt()->queryString); // SELECT * FROM table WHERE CreationDate IN (?, ?)
48  var_dump($e->getStmt()->preview()); // SELECT * FROM table WHERE CreationDate IN ('2013-01-01', '2013-06-01')
49 }
50 ```
51 
52 - You no longer need to deal with PDO::PARAM_INT, PDO::PARAM_STRING etc. since the PDO type is automatically set depending on the PHP type.
53 
54 - Each query you submit to the prepare() and sql() methods are hashed and stored for later re-use. Example :
55 
56 ```php
57 $cnx->sqlArray("SELECT * WHERE foo = ?", 'bar'); // "SELECT * WHERE foo = ?" string is prepared and transformed to a PDOStatementExtended object
58 $cnx->sqlArray("SELECT * WHERE foo = ?", 'baz'); // "SELECT * WHERE foo = ?" now matches an existing PDOStatementExtended object previously stored and doesn't have to be prepared again
59 ```
60 
61 - This behaviour may be disabled or enabled at any time (enabled by default) :
62 
63 ```php
64 $cnx->storeStmts(false); // or true
65 ```
66 
67 - You can also import an existing PDO instance, avoiding you to have a 2nd connection :
68 
69 ```php
70 $pdo = new \PDO('mysql:host=localhost', 'user', 'password');
71 $cnx = PDOExtended\PDOExtended::NewInstanceFromPdo($pdo);
72 ```

74 Full Example usage :
75 
76 ```php
77 
78 define('PDO_DSN', 'mysql:host=localhost;dbname=test');
79 define('PDO_USERNAME', 'root');
80 define('PDO_PASSWORD', null);
81 
82 // Normal call
83 $cnx = new PDOExtended(PDO_DSN, PDO_USERNAME, PDO_PASSWORD);
84 
85 // Let's create our working table...
86 $cnx->sql("CREATE TABLE IF NOT EXISTS
87  TVSeries (
88  Id TINYINT(3) UNSIGNED NOT NULL AUTO_INCREMENT,
89  Name VARCHAR(255),
90  Channel VARCHAR(40),
91  PRIMARY KEY (Id),
92  UNIQUE (Name)
93  ) ENGINE=InnoDb");
94 
95 // Example of insert with a prepared statement
96 $insertStmt = $cnx->Prepare("INSERT IGNORE INTO TVSeries SET Name = ?");
97 $tvSeries = Array('Games Of Thrones', 'The Big Bang Theory', 'Dexter');
98 
99 // For each Series, we play the insert statement with a different value
100 foreach ($tvSeries AS $series)
101  $insertStmt->sql($series);
102  // $series is mapped to the "Name" bound value - since it's a string, it'll be bound as PDO::PARAM_STR
103  // If $series was an integer, it'd be bound as PDO::PARAM_INT automatically
104 
105 // The expected parameter may also be an array, in case you have more values to bind
106 $insertStmt = $cnx->Prepare("INSERT IGNORE INTO TVSeries SET Name = :Name");
107 foreach ($tvSeries AS $series)
108  $insertStmt->sql(Array('Name' => $series));
109 
110 // Now, let's update the previous rows with the Channel info.
111 $channels = Array('HBO', 'CBS', 'ShowTime');
112 
113 // You're not obliged to prepare your statement before executing them. It's the sql() method's job.
114 $cnx->sql("UPDATE TVSeries SET Channel = ? WHERE Name = ?", Array($channels[0], $tvSeries[0]));
115 $cnx->sql("UPDATE TVSeries SET Channel = :Channel WHERE Name LIKE :Name", Array('Channel' => $channels[1], 'Name' => $tvSeries[1]));
116 $cnx->sql("UPDATE TVSeries SET Channel = :Channel WHERE Id = :Id", Array('Channel' => $channels[2], 'Id' => 3)); // Id will be cast as PDO::PARAM_INT because 2 is a PHP integer
117 
118 // Now, let's easily retrieve some infos.
119 
120 // All the table into a multidimensionnal array.
121 var_dump($cnx->sqlArray("SELECT * FROM TVSeries"));
122 
123 // Juste the Big bang theory row
124 var_dump($cnx->sqlRow("SELECT * FROM TVSeries WHERE Id = ?", 2));
125 
126 // A list of series
127 var_dump($cnx->sqlColumn("SELECT Name FROM TVSeries WHERE Channel IN (". PDOStatementExtended::PlaceHolders($channels) .")", $channels)); // PlaceHolders function will output "IN (?, ?, ?)";
128 
129 // What's the channel of Dexter ?
130 var_dump($cnx->sqlValue("SELECT Channel FROM TVSeries WHERE Name LIKE :Name", Array('Name' => 'Dexter')));
131 
132 // Another way to request it
133 var_dump($cnx->sqlValue("SELECT Channel FROM TVSeries WHERE Name LIKE ?", Array('Dexter')));
134 
135 // Another way to request it
136 var_dump($cnx->sqlValue("SELECT Channel FROM TVSeries WHERE Name LIKE ?", 'Dexter'));
137 
138 // Association key => value
139 var_dump($cnx->sqlAssoc("SELECT Channel, Name FROM TVSeries WHERE Id = ?", 3));
140 
141 // Association key => associative array
142 var_dump($cnx->sqlAssoc("SELECT Channel, Id, Name FROM TVSeries WHERE Id = ?", 3, PDOExtended::TO_ARRAY_ASSOC));
143 
144 // Association key => indexed array
145 var_dump($cnx->sqlAssoc("SELECT Channel, Id, Name FROM TVSeries WHERE Id = ?", 3, PDOExtended::TO_ARRAY_INDEX));
146 
147 // Association key => stdClass
148 var_dump($cnx->sqlAssoc("SELECT Channel, Id, Name FROM TVSeries WHERE Id = ?", 3, PDOExtended::TO_STDCLASS));
149 
150 // Association key => value, multiline version (array of keys => values)
151 var_dump($cnx->sqlMultiAssoc("SELECT Channel, Name FROM TVSeries WHERE Id IN (?, ?)", Array(1, 2)));
152 
153 // Association key => associative array, multiline version (array of keys => associative arrays)
154 var_dump($cnx->sqlMultiAssoc("SELECT Channel, Id, Name FROM TVSeries WHERE Id IN (?, ?)", Array(1, 2), PDOExtended::TO_ARRAY_ASSOC));
155 
156 // Association key => indexed array, multiline version (array of keys => indexed arrays)
157 var_dump($cnx->sqlMultiAssoc("SELECT Channel, Id, Name FROM TVSeries WHERE Id IN (?, ?)", Array(1, 2), PDOExtended::TO_ARRAY_INDEX));
158 
159 // Association key => stdClass, multiline version (array of keys => stdClasses)
160 var_dump($cnx->sqlMultiAssoc("SELECT Channel, Id, Name FROM TVSeries WHERE Id IN (?, ?)", Array(1, 2), PDOExtended::TO_STDCLASS));
161 
162 // You can also invoke all theses methods from a PDOStatementExtended object
163 var_dump($cnx->prepare("SELECT Channel, Id, Name FROM TVSeries WHERE Id IN (?, ?)")->sqlMultiAssoc(Array(2, 3), PDOExtended::TO_STDCLASS));
164 
165 // How long the query has taken ?
166 $stmt = $cnx->prepare("SELECT * FROM TVSeries WHERE Id = :Id OR Name LIKE :Name");
167 $res = $stmt->sqlArray(Array('Id' => 1, 'Name' => 'Dexter'));
168 var_dump($stmt->getDuration());
169 
170 // What was the real query played ?
171 var_dump($stmt->debug()->preview());
172 
173 // You can disconnect : every call afterwards will result in a PDO Exception until you invoke the Reconnect() method
174 $cnx->disconnect();
175 
176 // You can also Pause the connection. It actually disconnects from MySQl, but on the next call (sql, query, sqlArray etc) the connect() method will automatically be called so you never have a "not connected" exception thrown
177 $cnx->pause();
178 
179 // Why doing this ? When you have a big treatment to do (parsing a big xml for instance), this prevents from getting "sleep connections" issues
180 
181 
182 // Every other PDO method is available... $cnx->query() ou $cnx->setAttribute(), etc.
